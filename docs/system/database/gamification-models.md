# Gamification Models

> Quests, gifts, scenes, memories, achievements, and daily rewards.
> Last updated: 2026-04-09 · Reference: `server/prisma/schema.prisma`

## Quest System

```prisma
model Quest {
  id              String        @id @default(uuid())
  title           String
  description     String
  type            QuestType
  category        String        // daily, relationship, special, story
  requirements    Json          // { action: "send_message", count: 5 }
  rewardXp        Int           @default(0)
  rewardCoins     Int           @default(0)
  rewardGems      Int           @default(0)
  rewardAffection Int           @default(0)
  rewardItems     String[]      @default([])
  unlockLevel     Int           @default(1)
  requiresPremium Boolean       @default(false)
  minimumTier     PremiumTier   @default(FREE)
  sortOrder       Int           @default(0)
  isActive        Boolean       @default(true)
  startsAt        DateTime?
  endsAt          DateTime?
}

model UserQuest {
  id          String      @id @default(uuid())
  userId      String
  questId     String
  progress    Int         @default(0)
  maxProgress Int         @default(1)
  status      QuestStatus @default(IN_PROGRESS)
  startedAt   DateTime    @default(now())
  completedAt DateTime?
  claimedAt   DateTime?

  @@unique([userId, questId])
}
```

### Quest Enums

```prisma
enum QuestType { DAILY, WEEKLY, STORY, ACHIEVEMENT, EVENT, RELATIONSHIP }
enum QuestStatus { IN_PROGRESS, COMPLETED, CLAIMED, EXPIRED }
```

## Gift System

```prisma
model Gift {
  id              String        @id @default(uuid())
  name            String
  description     String
  emoji           String        @default("🎁")
  imageUrl        String
  category        String        // flower, jewelry, food, experience, special
  rarity          Rarity        @default(COMMON)
  requiresPremium Boolean       @default(false)
  minimumTier     PremiumTier   @default(FREE)
  priceCoins      Int           @default(0)
  priceGems       Int           @default(0)
  affectionBonus  Int           @default(10)
  unlockLevel     Int           @default(1)
  isLimited       Boolean       @default(false)
  availableFrom   DateTime?
  availableUntil  DateTime?
}

model UserGift {
  id        String   @id @default(uuid())
  userId    String
  giftId    String
  quantity  Int      @default(1)

  @@unique([userId, giftId])
}

model GiftHistory {
  id          String   @id @default(uuid())
  userId      String   // Sender
  characterId String
  giftId      String
  message     String?
  reaction    String?  // AI's reaction
  createdAt   DateTime @default(now())
}
```

### Rarity (5 tiers)

```prisma
enum Rarity { COMMON, UNCOMMON, RARE, EPIC, LEGENDARY }
```

## Scene System

```prisma
model Scene {
  id               String          @id @default(uuid())
  name             String          @unique
  description      String
  imageUrl         String
  category         String          // indoor, outdoor, romantic, adventure, school, work
  ambiance         String          // cozy, romantic, energetic, peaceful
  unlockMethod     String          @default("level")  // level, purchase, quest, event, relationship
  unlockValue      Int             @default(1)
  requiredStage    RelationshipStage? @default(STRANGER)
  priceGems        Int             @default(0)
  requiresPremium  Boolean         @default(false)
  minimumTier      PremiumTier     @default(FREE)
  isDefault        Boolean         @default(false)
}
```

## Memory System

model Memory {

## Achievement System

```prisma
model Achievement {
  id          String   @id @default(uuid())
  name        String   @unique
  description String
  category    String   // chat, gift, quest, relationship, special
  requirement Json     // { action: "total_messages", count: 100 }
  rewardXp    Int      @default(0)
  rewardCoins Int      @default(0)
  rewardGems  Int      @default(0)
  points      Int      @default(10)
  isSecret    Boolean  @default(false)
}

model UserAchievement {
  id            String   @id @default(uuid())
  userId        String
  achievementId String
  progress      Int      @default(0)
  unlockedAt    DateTime?
  claimedAt     DateTime?

  @@unique([userId, achievementId])
}
```

## Daily Rewards

```prisma
model DailyReward {
  id          String   @id @default(uuid())
  userId      String
  day         Int      // 1-7 for weekly cycle
  rewardType  String   // coins, gems, gift, xp
  rewardValue Int
  claimedAt   DateTime @default(now())

  @@index([userId, claimedAt])
  @@index([userId, day])
}
```
